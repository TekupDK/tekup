```yaml

name: Deploy Docs

  push:
    branches: [main]
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - run: pip install mkdocs-material
      - run: mkdocs build
      - run: mkdocs gh-deploy
```
